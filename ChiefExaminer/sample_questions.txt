name: Build Windows EXE


on:
  workflow_dispatch:


jobs:
  build:
    runs-on: windows-latest


    steps:
      - name: Download repository
        uses: actions/checkout@v4


      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.9'


      - name: Install PyInstaller
        run: |
          python -m pip install --upgrade pip
          pip install pyinstaller

      - name: Build EXE
        run: |
          cd ChiefeXam
          pyinstaller --onefile --name kyeremeh --add-data "public;public" --add-data "data;data" --add-data "schema.sql;." server.py
          
      - name: Upload EXE
        uses: actions/upload-artifact@v4
        with:
          name: kyeremeh-Windows-EXE
          path: ChiefExam/dist/kyeremeh.exe
